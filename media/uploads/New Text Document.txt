    contact: String,
    tickets: Number,
    totalAmount: Number,
    event: String,
    date: { type: Date, default: Date.now }
});

module.exports = mongoose.model('Booking', bookingSchema);

const subtractButton = document.getElementById("subtractTicket");
const addButton = document.getElementById("addTicket");
const ticketCountDisplay = document.getElementById("ticketCount");
const ticketTotalAmountDisplay = document.getElementById("ticketTotalAmount");
const finalTotalAmount = document.getElementById('charges1');

let ticketCount = 0;
const ticketPrice = 500;

function updateTotalAmount() {
    const totalAmount = ticketCount * ticketPrice;
    ticketTotalAmountDisplay.textContent = `₹${totalAmount}`;
    finalTotalAmount.textContent = `₹${totalAmount}`;
}

subtractButton.addEventListener("click", () => {
    if (ticketCount > 0) {
        ticketCount--;
        ticketCountDisplay.textContent = ticketCount;
        updateTotalAmount();
    }
});

addButton.addEventListener("click", () => {
    if (ticketCount < 500) { 
        ticketCount++;
        ticketCountDisplay.textContent = ticketCount;
        updateTotalAmount();
    }
});

updateTotalAmount();

async function openRazorpay() {
    const totalAmount = ticketCount * ticketPrice;

    if (totalAmount <= 0) {
        alert("Please select at least one ticket.");
        return;
    }

    const orderResponse = await fetch('/create-order', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({
            amount: totalAmount,
            currency: 'INR'
        })
    });

    const order = await orderResponse.json();

    const options = {
        key: "rzp_test_e9Zcka71inHlTQ",
        amount: order.amount,
        currency: order.currency,
        name: "A R Rahman Live in Concert",
        description: "Concert Ticket Booking",
        image: "https://example.com/your_logo", 
        order_id: order.id,
        handler: function (response) {
            verifyPayment(response, totalAmount);
        },
        prefill: {
            name: "Mahasakthivel", 
            email: "mahasakthivel0425@gmail.com",
            contact: "8220744842" 
        },
        theme: {
            color: "#DE5E27"
        }
    };

    const rzp = new Razorpay(options);
    rzp.open();
}

async function verifyPayment(response, totalAmount) {
    const verifyResponse = await fetch('/verify-payment', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({
            ...response,
            username: "Mahasakthivel", // Replace with actual user data
            email: "mahasakthivel0425@gmail.com", // Replace with actual user data
            contact: "8220744842", // Replace with actual user data
            tickets: ticketCount,
            totalAmount: totalAmount
        })
    });

    const result = await verifyResponse.json();

    if (result.status === 'success') {
        alert("Payment successful.");
    } else {
        alert("Payment verification failed.");
    }
}

const express = require('express');
const bodyParser = require('body-parser');
const Razorpay = require('razorpay');
const crypto = require('crypto');
const mongoose = require('mongoose');
const path = require('path');
const Booking = require('./models/Booking'); // Import the Booking model

const app = express();
const PORT = process.env.PORT || 3000;

app.use(bodyParser.json());
app.use(bodyParser.urlencoded({ extended: true }));

const razorpay = new Razorpay({
    key_id: 'rzp_test_e9Zcka71inHlTQ',
    key_secret: 'QSyMwnyKV70zGgzToZ5K6sgg'
});

// Connect to MongoDB
mongoose.connect('mongodb://localhost:27017/ticketBooking', { useNewUrlParser: true, useUnifiedTopology: true })
    .then(() => console.log('MongoDB connected...'))
    .catch(err => console.log(err));

app.use(express.static(path.join(__dirname, 'public')));

app.post('/create-order', async (req, res) => {
    const { amount, currency } = req.body;
    const options = {
        amount: amount * 100, // Amount in paise
        currency,
        receipt: `receipt_order_${Math.floor(Math.random() * 10000)}`
    };

    try {
        const order = await razorpay.orders.create(options);
        res.json(order);
    } catch (error) {
        res.status(500).send(error);
    }
});

app.post('/verify-payment', async (req, res) => {
    const { razorpay_order_id, razorpay_payment_id, razorpay_signature, username, email, contact, tickets, totalAmount } = req.body;
    const hmac = crypto.createHmac('sha256', razorpay.key_secret);
    hmac.update(razorpay_order_id + '|' + razorpay_payment_id);
    const generated_signature = hmac.digest('hex');

    if (generated_signature === razorpay_signature) {
        // Save booking data to MongoDB
        const booking = new Booking({
            username,
            email,
            contact,
            tickets,
            totalAmount,
            event: 'A R Rahman Live in Concert'
        });
        await booking.save();

        res.json({ status: 'success' });
    } else {
        res.json({ status: 'failure' });
    }
});

app.listen(PORT, () => {
    console.log(`Server is running on port ${PORT}`);
});
